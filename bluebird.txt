function limparDiretorio($diretorio) {
    // Verifica se o diretório existe
    if (!is_dir($diretorio)) {
        echo "O diretório especificado não existe.";
        return;
    }

    // Obtém uma lista de todos os arquivos e pastas no diretório
    $itens = glob($diretorio . '/*');

    // Verifica se a lista de itens não está vazia
    if ($itens !== false) {
        // Itera sobre cada item
        foreach ($itens as $item) {
            // Verifica se é um arquivo
            if (is_file($item)) {
                // Se for um arquivo, exclui-o
                unlink($item);
            } elseif (is_dir($item)) {
                // Se for um diretório, chama a função novamente para limpar o diretório interno
                limparDiretorio($item);
                // Após limpar o diretório interno, exclui o diretório externo
                rmdir($item);
            }
        }
        echo "O diretório foi limpo com sucesso.";
    } else {
        echo "O diretório está vazio.";
    }
}

limparDiretorio('old/');
